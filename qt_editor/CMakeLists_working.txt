cmake_minimum_required(VERSION 3.16)
project(FastEngineWorkingEditor)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Find Qt6
find_package(Qt6 REQUIRED COMPONENTS Core Widgets)

# Enable Qt6 MOC
set(CMAKE_AUTOMOC ON)

# Working editor executable (без проблемных OpenGL компонентов)
add_executable(FastEngineWorkingEditor 
    WorkingEditor.cpp
)

# Link Qt6 libraries
target_link_libraries(FastEngineWorkingEditor 
    Qt6::Core
    Qt6::Widgets
)

# Set target properties
set_target_properties(FastEngineWorkingEditor PROPERTIES
    MACOSX_BUNDLE TRUE
    MACOSX_BUNDLE_BUNDLE_NAME "FastEngine Qt Editor"
    MACOSX_BUNDLE_BUNDLE_VERSION "1.0.0"
    MACOSX_BUNDLE_SHORT_VERSION_STRING "1.0.0"
    MACOSX_BUNDLE_IDENTIFIER "com.fastengine.qteditor"
)
